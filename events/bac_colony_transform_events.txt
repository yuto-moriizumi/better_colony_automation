namespace = colony_automation_event

# cancel arcology project candidate
planet_event = {
	id = colony_automation_event.100
	hide_window = yes
	is_triggered_only = yes
	trigger = { always = yes }
	immediate = {
		remove_modifier = bca_pm_arcology_project_candidate
		clear_open_selector = yes
		bca_secondary_zone_setting_set_by_current_zones = yes
	}
}

planet_event = {
	id = colony_automation_event.101
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_modifier = bca_pm_arcology_project_candidate
	}

	immediate = {
		# check if it no longer meets the criteria for being an arcology project candidate
		if = {
			limit = {
				NOR = {
					AND = {
						NOR = {
							is_planet_class = pc_city
							is_planet_class = pc_relic
							has_modifier = resort_colony
							has_modifier = penal_colony
							has_modifier = slave_colony
							has_modifier = pm_wenkwort_custodian
							has_designation = col_nanotech_world
						}
						exists = owner
						owner = { has_ascension_perk = ap_arcology_project }
					}
					AND = {
						is_planet_class = pc_relic
						exists = owner
						owner = {
							OR = {
								is_regular_empire = yes
								has_valid_civic = civic_machine_servitor
							}
						}
					}
				}
			}
			planet_event = { id=colony_automation_event.100 }
		}
		# city_world_candidate
		else_if = {
			limit = {
				bca_has_resource_for_arcology_project = yes
				OR = {
					AND = {
						free_district_slots < 1
						num_uncleared_blockers < 1
						num_districts = { type = district_farming value < 1 }
						num_districts = { type = district_mining value < 1 }
						num_districts = { type = district_generator value < 1 }
						# Volcanic Worlds
						num_districts = { type = district_geothermal value < 1 }
						num_districts = { type = district_melting value < 1 }
						num_districts = { type = district_polytechnic value < 1 }

						NOT = { has_planet_flag = legendary_leader_planet }
					}
					AND = {
						is_planet_class = pc_relic
						is_artificial = no
						exists = owner
						num_uncleared_blockers < 1
						owner = { has_technology = tech_housing_2 }
					}
				}
			}
			planet_event = { id = colony_automation_event.102 }

		}
	}

}

# 理想城建造询问事件
planet_event = {
	id = colony_automation_event.102
	title = colony_automation_event.102.name
	desc = colony_automation_event.102.desc
	picture = GFX_evt_metropolis
	show_sound = event_administrative_work
	location = ROOT

	is_triggered_only = yes

	option = {
		name = colony_automation_event.102.a
		if = {
			limit = { is_planet_class = pc_relic }
			select_decision = { name = decision_arcology_project_relic }
		}
		else = {
			select_decision = { name = decision_arcology_project }
		}
		hidden_effect = {
			remove_modifier = bca_pm_arcology_project_candidate
		}
		sound = event_administrative_work
	}

	option = {
		name = colony_automation_event.102.b
		hidden_effect = {
			planet_event = { id=colony_automation_event.100 }
		}
	}


}
