		}

		}

		}

			id = test_event.1
		}
	}

		name = "build check 2"
		add_district = { district_type = district_city }
		set_variable = {
			which = bca_test_district_metric_1
			value = $metric_value$
		}
		remove_district = district_city
		add_district = { district_type = district_generator }
		set_variable = {
			which = bca_test_district_metric_2
			value = $metric_value$
		}
		remove_district = district_generator
		if = {
			limit = {
				check_variable = {
					which = bca_test_district_metric_1
					value >= bca_test_district_metric_2
				}
			}
			set_planet_flag = bca_pf_has_district_build_plan
			set_planet_flag = bca_pf_plan_build_district_city
		}
		else = {
			set_planet_flag = bca_pf_has_district_build_plan
			set_planet_flag = bca_pf_plan_build_district_generator
		}
		planet_event = {
			id = test_event.1
		}
	}

	option = {
		name = "replace check"
		bca_test_district_to_replace_city_bidirectional = {
			district_2 = district_generator
			metric_value = value:bca_protential_standard_energy_production
		}
		planet_event = {
			id = test_event.1
		}
	}

	option = {
		name = "bca_pf_has_district_build_plan"

			has_planet_flag = bca_pf_has_district_build_plan
		}
		remove_planet_flag = bca_pf_has_district_build_plan

			id = test_event.1
		}

		}

		}

		}
		remove_planet_flag = bca_pf_plan_remove_district_city
		planet_event = {
			id = test_event.1
		}
	}
