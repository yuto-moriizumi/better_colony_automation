#---------------------------------
# Districts
#---------------------------------

bca_planet_max_districts_of = {
	modifier = {
		bca_is_uncapped_$type$ = no
		set = modifier:$type$_max_add
	}
	modifier = {
		bca_is_uncapped_$type$ = yes
		set = value:bca_planet_max_districts
	}
}

bca_resource_designation_num_city_districts = {
	complex_trigger_modifier = {
		trigger = num_districts
		parameters = {
			type = district_city
		}
		mode = add
	}
	complex_trigger_modifier = {
		trigger = num_districts
		parameters = {
			type = district_hive
		}
		mode = add
	}
	complex_trigger_modifier = {
		trigger = num_districts
		parameters = {
			type = district_nexus
		}
		mode = add
	}
}

bca_production_inc_mult_from_urban_resource = {
	base = 0.2
	multiply = value:bca_resource_designation_num_city_districts
}

bca_production_inc_jobs_from_zone= {
	base = 100
	multiply = value:bca_resource_designation_num_city_districts
}



#---------------------------------
# Energy
#---------------------------------

bca_potential_urban_energy_production_inc_mult = {
	set = $num_d0$
	mult = value:bca_future_planed_zone_urban_energy
	mult = 0.2
}

bca_planet_energy_job_mult_estimate = {
	# remove the modifier from zone_urban_energy
	complex_trigger_modifier = {
		trigger = num_zones
		parameters = {
			type = zone_urban_energy
		}
		mode = subtract
	}
	mult = value:bca_production_inc_mult_from_urban_resource
	add = value:bca_planet_energy_job_mult # current value is mult with out zone_urban_energy
	# assume 2 zone_urban_energy will be built, each city district provides 0.4 multiplier
	add = value:bca_potential_urban_energy_production_inc_mult|num_d0|$num_district_city$|
	modifier = {
		add = 0.2
		NOR = {
			has_building = building_energy_grid
			has_building = building_energy_nexus
		}
		bca_can_build_building_energy_grid = yes
	}
}

bca_planet_energy_job_add_estimate = {
	set = value:bca_planet_energy_job_add
	modifier = {
		add = 2
		NOT = {has_building = building_energy_nexus}
		bca_can_build_building_energy_nexus = yes
	}
}

bca_potential_standard_energy_production_estimate = {
	set = $NUM_R_DST$
	multiply = 300
	divide = 100
	mult = value:bca_planet_energy_job_add_estimate
	mult = value:bca_planet_energy_job_mult_estimate|num_district_city|$NUM_C_DST$|
}

bca_potential_energy_district_max_inc = {
	modifier = {
		has_building = building_generator_districts_1
		subtract = @b1_max_districts_add
	}
	modifier = {
		has_building = building_generator_districts_2
		subtract = @b2_max_districts_add
	}
	modifier = {
		has_building = building_generator_districts_3
		subtract = @b3_max_districts_add
	}
	modifier = {
		has_building = building_generator_districts_4
		subtract = @b4_max_districts_add
	}
	add = @b4_max_districts_add
}

bca_potential_district_max_add_energy = {
	add = modifier:district_generator_max_add
	add = modifier:district_hab_energy_max_add
	add = modifier:district_photosynthesis_fields_max_add
	modifier = {
		bca_can_build_building_generator_districts_1 = yes
		add = value:bca_potential_energy_district_max_inc
	}
	modifier = {
		OR = {
			is_generator_district_uncapped = yes
			bca_has_secondary_districts = yes
		}
		set = value:bca_planet_max_districts
	}
	max = value:bca_planet_max_districts
}
