bca_potential_max_districts = {
	complex_trigger_modifier = {
		trigger = num_districts
		parameters = {
			type = any
		}
		mode = set
	}
	complex_trigger_modifier = {
		trigger = num_free_districts
		parameters = {
			type = any
		}
		mode = add
	}
    modifier = {
        has_modifier = bca_pm_arcology_project_candidate
        add = trigger:planet_size
    }
}

bca_num_free_secondary_districts_plan = {
    set = value:bca_potential_max_districts
    subtract = value:num_primary_district
    subtract = bca_ps_plan_district_count_z3
    subtract = bca_ps_plan_district_count_z4
    subtract = bca_ps_plan_district_count_z5
}

bca_num_free_primary_districts_plan = {
    add = bca_ps_plan_district_count_z3
    add = bca_ps_plan_district_count_z4
    add = bca_ps_plan_district_count_z5
    subtract = value:num_secondary_district
}
