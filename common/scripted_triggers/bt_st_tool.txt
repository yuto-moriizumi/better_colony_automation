any_owned_pop_group = {
	OR = {
		is_pop_category = civilian
		is_pop_category = maintenance_drone
	}
}


requires_more_job = {
	OR = {
		has_planet_flag = bca_pf_planet_pre_construction
		free_jobs <= owner.bca_reserve_job_amount
	}
}
really_requires_more_job = {
	free_jobs <= owner.bca_reserve_job_amount
}

bca_has_minerals_to_build = {
	if = {
		limit = { is_scope_type = planet }
		owner = {
			resource_stockpile_compare = { resource = minerals value >= bca_reserve_minerals_amount }
		}
	}
	else = {
		resource_stockpile_compare = { resource = minerals value >= bca_reserve_minerals_amount }
	}
}

bca_has_resource_for_arcology_project = {
	owner = {
		resource_stockpile_compare = {
			resource = alloys
			value >= value:bca_value_add|VALUE|bca_reserve_alloys_amount|ADD|10000|
		}
		resource_stockpile_compare = {
			resource = influence 
			value >= value:bca_value_add|VALUE|bca_reserve_influence_amount|ADD|200|
		}
	}
}
