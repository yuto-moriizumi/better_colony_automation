bca_can_remove_building = {
	AND = {
		NOT = { has_planet_flag = bca_pf_disable_planet_auto_destruction_building }
	}
}

bca_can_remove_pop_assemble = {
	AND = {
		bca_can_remove_building = yes
		owner = { has_country_flag = bac_remove_pop_assemble_buildings }
		free_district_slots = 0
		free_building_slots = 0
		free_housing <= 0
		free_jobs <= 0
	}
}

bca_can_build_pop_assemble = {
	free_jobs > 0
	free_building_slots > 1
}

bca_unique_building_potential = {
	bca_can_build_$building1$ = yes
	NOR = {
		has_building = $building1$
		[[building2] has_building = $building2$ ]
		[[building3] has_building = $building3$ ]
		[[building4] has_building = $building4$ ]
	}
}
