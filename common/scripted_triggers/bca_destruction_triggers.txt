bca_can_remove_building = {
	AND = {
		owner = { has_country_flag = bac_building_destroy }
		NOT = { has_modifier = bca_pm_disable_planet_auto_destruction }
	}
}


bca_can_remove_building_reasearch = {
	AND = {
		bca_can_remove_building = yes
		free_building_slots = 0
		has_any_research_designation = no
	}
}

bca_can_remove_building_factory = {
	AND = {
		bca_can_remove_building = yes
		free_building_slots = 0
		has_factory_designation = no
		has_industrial_designation = no
	}
}

bca_can_remove_building_foundry = {
	AND = {
		bca_can_remove_building = yes
		free_building_slots = 0
		has_foundry_designation = no
		has_industrial_designation = no
	}
}

bca_can_remove_building_industry = {
	AND = {
		bca_can_remove_building = yes
		free_building_slots = 0
		has_any_industrial_designation = no
	}
}

bca_can_remove_building_trade = {
	AND = {
		bca_can_remove_building = yes
		free_building_slots = 0
		bca_has_any_trade_designation = no
	}
}

bca_can_remove_pop_assemble = {
	AND = {
		bca_can_remove_building = yes
		owner = { has_country_flag = bac_remove_pop_assemble_buildings }
		free_district_slots = 0
		free_building_slots = 0
		free_housing <= 0
		free_jobs <= 0
	}
}

bca_can_build_pop_assemble = {
	free_jobs > 0
	free_building_slots > 1
}

bca_can_remove_district_generator = {
	OR = {
		AND = {
			free_district_slots = 0
			has_modifier = bca_pm_arcology_project_candidate
		}
		has_planet_flag = bca_pf_plan_remove_district_generator
	}
}

bca_can_remove_district_mining = {
	OR = {
		AND = {
			free_district_slots = 0
			has_modifier = bca_pm_arcology_project_candidate
		}
		has_planet_flag = bca_pf_plan_remove_district_mining
	}
}

bca_can_remove_district_farming = {
	OR = {
		AND = {
			free_district_slots = 0
			has_modifier = bca_pm_arcology_project_candidate
		}
		has_planet_flag = bca_pf_plan_remove_district_farming
	}
}



bca_can_remove_district_hab_housing = {
	AND = {
		free_district_slots = 0
		free_housing > 700
		OR = {
			AND = {
				has_generator_designation = yes
				num_free_districts = { type = district_hab_energy value = 0 }
			}
			AND = {
				has_mining_designation = yes
				num_free_districts = { type = district_hab_mining value = 0 }
			}
			AND = {
				has_any_research_designation = yes
				num_free_districts = { type = district_hab_science value = 0 }
			}
		}
	}
}



bca_district_reach_planet_max = {
	check_modifier_value = {
		modifier = $type$_max_add
		value = value:bca_num_districts_value|TYPE|$type$|
	}
}

bca_require_district_generator = {
	has_planet_flag = bca_pf_plan_build_district_generator
}
bca_require_district_mining = {
	has_planet_flag = bca_pf_plan_build_district_mining
}

bca_require_district_farming = {
	has_planet_flag = bca_pf_plan_build_district_farming
}


bca_can_remove_district_city = {
	OR = {
		has_planet_flag = bca_pf_plan_remove_district_city
		has_planet_flag = bca_pf_plan_remove_district_hive
		has_planet_flag = bca_pf_plan_remove_district_nexus
	}
}

bca_require_district_city = {
	has_planet_flag = bca_pf_has_district_build_plan
	OR = {
		has_planet_flag = bca_pf_plan_build_district_city
		has_planet_flag = bca_pf_plan_build_district_hive
		has_planet_flag = bca_pf_plan_build_district_nexus
	}
}

bca_require_district_rw_farming = {
	AND = {
		has_designation = col_ring_farming
		num_districts = { type = district_rw_farming value = 0 }
	}
}
bca_require_district_rw_generator = {
	AND = {
		has_designation = col_ring_generator
		num_districts = { type = district_rw_generator value = 0 }
	}
}

bca_can_remove_district_rw_city = {
	AND = {
		free_district_slots = 0
		free_housing > 1500
		OR = {
			bca_require_district_rw_farming = yes
			bca_require_district_rw_generator = yes
		}
	}
}
