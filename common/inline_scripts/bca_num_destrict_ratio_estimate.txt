if = {
	limit = {
		OR = {
			check_variable = {
				which = opti_cond_planet_modifier_job_add
				value != value:$FACTOR_JOB_ADD$
			}
			check_variable = {
				which = opti_cond_planet_modifier_job_mult
				value != value:$FACTOR_JOB_MULT$
			}
			check_variable = {
				which = opti_cond_opti_cond_max_districts_add
				value != value:bca_planet_max_districts
			}
			check_variable = {
				which = opti_cond_$DISTRICT$_max_add
				value != value:$PROTENTIAL_DISTRICT_MAX$
			}
		}
	}
	# mark the condition that optimal number of generator districts is found
	set_variable = {
		which = opti_cond_planet_modifier_job_add
		value = value:$FACTOR_JOB_ADD$
	}
	set_variable = {
		which = opti_cond_planet_modifier_job_mult
		value = value:$FACTOR_JOB_MULT$
	}
	set_variable = {
		which = opti_cond_opti_cond_max_districts_add
		value = value:bca_planet_max_districts
	}
	set_variable = {
		which = opti_cond_$DISTRICT$_max_add
		value = value:$PROTENTIAL_DISTRICT_MAX$
	}
	# start with all district to be energy districts
	if = {
		limit = {
			check_variable_arithmetic = {
				which = value:$PROTENTIAL_DISTRICT_MAX$
				subtract = 1
				value >= value:bca_planet_max_districts
			}
		}
		set_variable = {
			which = sim_resource_district_count
			value = value:bca_planet_max_districts
		}
		set_variable = {
			which = sim_district_city_count
			value = 1
		}
		subtract_variable = {
			which = sim_resource_district_count
			value = sim_district_city_count
		}
	}
	else = {
		set_variable = {
			which = sim_resource_district_count
			value = value:$PROTENTIAL_DISTRICT_MAX$
		}
		set_variable = {
			which = sim_district_city_count
			value = value:bca_planet_max_districts
		}
		subtract_variable = {
			which = sim_district_city_count
			value = sim_resource_district_count
		}
	}


	set_variable = {
		which = bca_opti_count_$DISTRICT$
		value = sim_resource_district_count
	}
	set_variable = {
		which = bca_opti_count_primary_district
		value = sim_district_city_count
	}

	set_variable = {
		which = max_production
		value = 0
	}

	set_variable = {
		which = production
		value = value:$PRODUCTION_ESTIMATION$|NUM_R_DST|sim_resource_district_count|NUM_C_DST|sim_district_city_count|
	}

	while = {
		limit = {
			check_variable = {
				which = production
				value > max_production
			}
			check_variable = {
				which = sim_resource_district_count
				value > 1
			}
		}

		set_variable = {
			which = max_production
			value = production
		}
		set_variable = {
			which = bca_opti_count_$DISTRICT$
			value = sim_resource_district_count
		}
		set_variable = {
			which = bca_opti_count_primary_district
			value = sim_district_city_count
		}

		add_variable = {
			which = sim_district_city_count
			value = 1
		}
		subtract_variable = {
			which = sim_resource_district_count
			value = 1
		}

		set_variable = {
			which = production
			value = value:$PRODUCTION_ESTIMATION$|NUM_R_DST|sim_resource_district_count|NUM_C_DST|sim_district_city_count|
		}
	}

	set_variable = {
		which = debug_max_production
		value = max_production
	}
	set_variable = {
		which = debug_job_mult_estimate
		value = value:bca_planet_energy_job_mult_estimate|num_district_city|sim_district_city_count|
	}
	set_variable = {
		which = debug_job_add_estimate
		value = value:bca_planet_energy_job_add_estimate
	}

	# retain bca_opti_count_$DISTRICT$ and bca_opti_count_primary_district and conditions
	clear_variable = sim_district_city_count
	clear_variable = sim_resource_district_count
	clear_variable = max_production
	clear_variable = production
}
