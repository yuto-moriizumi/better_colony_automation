
try_construct_district = {
	if = {
		limit = { check_variable = { which = bca_selected_index value = $index$ } }
		set_planet_flag = bca_pf_plan_build_district_$district_slot$
	}
	else_if = {
		limit = {
			NOT = { has_planet_flag = bca_pf_disable_auto_district_manage }
			check_variable = { which = bca_selected_index value = -1 }
			check_variable_arithmetic = {
				which = value:num_free_district_$district_slot$
				value > 0
			}

			[[district_max]
			check_variable_arithmetic = {
				which = value:num_district_$district_slot$
				value < $district_max$
			}
			]
		}
		bca_add_potential_district_$district_slot$ = yes
		bca_test_update_construction_metric_max_with_cost = {
			production = $production$
			cost = $cost$
			priority_shift = $priority_shift$
			index = $index$
		}
		bca_remove_district_$district_slot$ = yes
	}
}

try_construct_building = {
	if = {
		limit = { check_variable = { which = bca_selected_index value = $index$ } }
		bca_set_building_build_flag = {
			building = $building$
		}
	}
	else_if = {
		limit = { 
			check_variable = { which = bca_selected_index value = -1 } 
			bca_can_build_$building$ = yes
		}
		bca_test_update_construction_metric_max_with_cost = {
			[[!MULT]
			production = value:$standard_production$|INC_JOB|$INC_JOB$|INC_ADD|$INC_ADD$|INC_MULT|$INC_MULT$|
			]
			[[MULT]
			production = value:$standard_production$|INC_JOB|$INC_JOB$|INC_ADD|$INC_ADD$|INC_MULT|$INC_MULT$|MULT|$MULT$|
			]
			cost = $cost$
			priority_shift = $priority_shift$
			index = $index$
		}
	}
}


try_construct_zone_city = {
	if = {
		limit = { check_variable = { which = bca_selected_index value = $index$ } }
		set_planet_flag = bca_pf_plan_build_primary_zone_type_$zone_type$
	}
	else_if = {
		limit = { 
			check_variable = { which = bca_selected_index value = -1 }
			bca_can_build_$zone$ = yes
			num_zones = {
				type = $zone$
				value < value:bca_future_planed_$zone$
			}
		}
		bca_get_num_free_city_zones = yes

		if = {
			limit = { 
				check_variable = {
					which = bca_num_free_city_zones
					value > 0
				}
			}

			bca_test_update_construction_metric_max_with_cost = {
				production = $production$
				cost = $cost$
				priority_shift = $priority_shift$
				index = $index$
			}
		}


	}
}
