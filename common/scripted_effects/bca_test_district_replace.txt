bca_test_district_to_replace = {
	if = {
		limit = {
            check_variable_arithmetic = {
                which = value:num_district_$district_slot_replace_from$
                value > 1
            }
			bca_district_$district_slot_replace_to$_reach_planet_max = no
		}
		set_variable = {
			which = bca_test_district_metric_1
			value = $metric_value$
		}
		bca_remove_district_$district_slot_replace_from$ = yes
		bca_add_district_$district_slot_replace_to$ = yes
		set_variable = {
			which = bca_test_district_metric_2
			value = $metric_value$
		}
		bca_remove_district_$district_slot_replace_to$ = yes
		bca_add_district_$district_slot_replace_from$ = yes
		if = {
			limit = {
                if = {
                    limit = { always = $is_priority_district$ }
                    check_variable = {
                        which = bca_test_district_metric_2
                        value >= bca_test_district_metric_1
                    }
                }
                else = {
                    check_variable = {
                        which = bca_test_district_metric_2
                        value > bca_test_district_metric_1
                    }
                }
			}
			set_planet_flag = bca_pf_plan_remove_district_$district_slot_replace_from$
			set_planet_flag = bca_pf_plan_build_district_$district_slot_replace_to$
			set_planet_flag = bca_pf_has_district_remove_plan
		}

	}
}
