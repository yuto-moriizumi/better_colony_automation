


bca_opti_district_estimation_energy = {

    set_variable = {
        which = bca_opti_district_condition_check_sum_current
        value = value:bca_planet_energy_job_add
    }
    multiply_variable = {
        which = bca_opti_district_condition_check_sum_current
        value = value:bca_planet_energy_job_mult
    }
    multiply_variable = {
        which = bca_opti_district_condition_check_sum_current
        value = value:bca_potential_district_max_add_energy
    }
    multiply_variable = {
        which = bca_opti_district_condition_check_sum_current
        value = value:bca_planet_max_districts
    }

    if = {
        limit = {
            check_variable = {
                which = bca_opti_district_condition_check_sum
                value != bca_opti_district_condition_check_sum_current
            }
        }

        set_variable = {
            which = bca_opti_district_condition_check_sum
            value = bca_opti_district_condition_check_sum_current
        }


        if = {
            limit = {
                check_variable_arithmetic = {
                    which = value:bca_potential_district_max_add_energy
                    value >= value:bca_planet_max_districts
                }
            }
            set_variable = {
                which = sim_resource_district_count
                value = value:bca_planet_max_districts
            }
            set_variable = {
                which = sim_district_city_count
                value = 1
            }
            subtract_variable = {
                which = sim_resource_district_count
                value = sim_district_city_count
            }
        }
        else = {
            set_variable = {
                which = sim_resource_district_count
                value = value:bca_potential_district_max_add_energy
            }
            set_variable = {
                which = sim_district_city_count
                value = value:bca_planet_max_districts
            }
            subtract_variable = {
                which = sim_district_city_count
                value = sim_resource_district_count
            }
        }

        set_variable = {
            which = bca_opti_count_energy
            value = sim_resource_district_count
        }
        set_variable = {
            which = bca_opti_count_primary_district
            value = sim_district_city_count
        }

        set_variable = {
            which = max_production
            value = 0
        }

        set_variable = {
            which = production
            value = value:bca_potential_standard_energy_production_estimate|NUM_R_DST|sim_resource_district_count|NUM_C_DST|sim_district_city_count|
        }

        while = {
            limit = {
                check_variable = {
                    which = production
                    value > max_production
                }
                check_variable = {
                    which = sim_resource_district_count
                    value > 1
                }
            }

            set_variable = {
                which = max_production
                value = production
            }
            set_variable = {
                which = bca_opti_count_secondary_district
                value = sim_resource_district_count
            }
            set_variable = {
                which = bca_opti_count_primary_district
                value = sim_district_city_count
            }

            add_variable = {
                which = sim_district_city_count
                value = 1
            }
            subtract_variable = {
                which = sim_resource_district_count
                value = 1
            }

            set_variable = {
                which = production
                value = value:bca_potential_standard_energy_production_estimate|NUM_R_DST|sim_resource_district_count|NUM_C_DST|sim_district_city_count|
            }
        }

        set_variable = {
            which = debug_max_production
            value = max_production
        }
        set_variable = {
            which = debug_job_mult_estimate
            value = value:bca_planet_energy_job_mult_estimate|num_district_city|sim_district_city_count|
        }
        set_variable = {
            which = debug_job_add_estimate
            value = value:bca_planet_energy_job_add_estimate
        }



        clear_variable = sim_district_city_count
        clear_variable = sim_resource_district_count
        clear_variable = max_production
        clear_variable = production
    }
    clear_variable = bca_opti_district_condition_check_sum_current
}
