automate_rare_deficit_management_archaeo_refinery = {
	available = {
		free_building_slots > 0
		bca_has_minerals_to_build = yes
		has_any_industrial_designation = yes
		OR = {
			has_planet_flag = bca_pf_pre_build_rare_resource_buildings
			owner = { 
				OR = {
					has_monthly_income = { resource = rare_crystals value < 0 } 
					has_monthly_income = { resource = volatile_motes value < 0 }
					has_monthly_income = { resource = exotic_gases value < 0 }
				}
			}
		}
	}

	empire_wide_emergency = yes
	category = "planet_automation_rare_resources"
	buildings = {
		1 = {
			building = building_archaeo_refinery
		}
	}

}
