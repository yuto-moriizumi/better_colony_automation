				NOT = { has_forbidden_jobs = healthcare }

			}
		}

		holo = {
			building = building_holo_theatres
			available = {
				NOT = { has_available_jobs = entertainer }

				NOT = { has_forbidden_jobs = entertainer }
				NOT = { has_forbidden_jobs = healthcare }
