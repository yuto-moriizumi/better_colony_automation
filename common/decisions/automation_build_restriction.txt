decision_enable_pre_construction = {
	sound = event_administrative_work
	icon = decision_arcology_project

	resources = {
		category = decisions
	}

	potential = {
		exists = owner
		NOT = { has_modifier = planet_pre_construction }
	}

	effect = {
		add_modifier = { modifier = "planet_pre_construction" days = -1 }
	}
}

decision_disable_pre_construction = {
	sound = event_administrative_work
	icon = decision_arcology_project

	resources = {
		category = decisions
	}

	potential = {
		exists = owner
		has_modifier = planet_pre_construction
	}

	effect = {
		remove_modifier = "planet_pre_construction"
	}
}
