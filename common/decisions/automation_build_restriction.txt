decision_bca_enable_arcology_project_candidate = {
	sound = event_administrative_work
	icon = decision_arcology_project_candidate

	resources = {
		category = decisions
	}

	potential = {
		exists = owner
		NOT = { has_modifier = bca_pm_arcology_project_candidate }
		is_artificial = no
		exists = owner
		OR = {
			AND = {
				NOR = {
					is_planet_class = pc_city
					is_planet_class = pc_relic
					has_modifier = resort_colony
					has_modifier = penal_colony
					has_modifier = slave_colony
					has_modifier = pm_wenkwort_custodian
				}
				owner = { has_ascension_perk = ap_arcology_project }
				NOT = { has_designation = col_nanotech_world }
			}
			AND ={
				is_planet_class = pc_relic
				owner = {
					OR = {
						is_regular_empire = yes
						has_valid_civic = civic_machine_servitor
					}
				}
			}
		}

	}

	# 禁止在一产规划上启用
	allow = {
		has_farming_designation = no
		has_mining_designation = no
		has_generator_designation = no
	}

	effect = {
		add_modifier = { modifier = "bca_pm_arcology_project_candidate" days = -1 }
		remove_planet_flag = bca_pf_can_build_extra_zones_for_storage
	}
}
