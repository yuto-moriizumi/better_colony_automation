bca_retain_low_level_rare_resource_buildings = {
    potential = {
        has_technology = tech_archaeo_refinery
    }
    allow = { always = yes }
    option = {
        name = "bca_option_remove_low_level_rare_resource_buildings"
        on_enabled = {
            remove_country_flag = bca_retain_low_level_rare_resource_buildings
        }
        ai_weight = {
            factor = 1
        }
    }
    option = {
        name = "bca_option_retain_low_level_rare_resource_buildings"
        on_enabled = {
            set_country_flag = bca_retain_low_level_rare_resource_buildings
        }
        ai_weight = {
            factor = 0
        }
    }
}
